import sys


def add_data_for_name(name_data, year, rank, name):
    """ 
    :param name_data: dict, key=name, value: dict, dict{'year': rank, 'year': rank } 
    :param year: str, year 
    :param rank: str, rank in a year 
    :param name: baby's name 
    :return: no 
    """
    # check if name in name_data
    if name in name_data:
        # if already has rank in a year, change the rank to a smaller one
        if year in name_data[name]:
            if int(rank) < int(name_data[name][year]):
                name_data[name][year] = rank
        # if year is not added, add the year->rank pair
        else:
            name_data[name][year] = rank
    else:
        name_data[name] = {}
        name_data[name][year] = rank

def test1():
    name_data = {'Kylie': {'2010': '57'}, 'Nick': {'2010': '37'}}
    add_data_for_name(name_data, '2010', '208', 'Kate')
    print('--------------------test1----------------------')
    print(str(name_data))
    print('-----------------------------------------------')
